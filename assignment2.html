const int buttonPin = 2;  // the number of the pushbutton pin
const int yellowLed = 4;   // yellow led connected to pin 4
const int dimLed = 5;      // green led connected to pin 5
int buttonValue = 0;       // variable for reading the pushbutton status
int timer = 500;           // the higher the number, the slower the speed of blinking of yellow LED

// the setup function runs once when you press reset or power the board
void setup() {
  pinMode(yellowLed, OUTPUT); // initialize yellow LED pin as an output
  pinMode(buttonPin, INPUT); // initialize pushbutton pin as an input
}

void loop() {
  buttonValue = digitalRead(buttonPin); //read the state of the pushbutton value:
  if (buttonValue == HIGH) { //check if the pushbutton is pressed. If pressed, buttonValue is High.
    digitalWrite(yellowLed, HIGH); //turn on yellow LED
    delay(timer); //wait for 0.5 seconds
    digitalWrite(yellowLed, LOW);  //turn off yellow LED
    for (int fadeValue = 0 ; fadeValue <= 255; fadeValue += 5) { //fade in from min to max in increments of 5 points
    analogWrite(dimLed, fadeValue); //sets the value (range from 0 to 255):
    delay(20); // wait for 20 milliseconds to see the dimming effect
    }
    
    for (int fadeValue = 255 ; fadeValue >= 0; fadeValue -= 5) {  // fade out from max to min in increments of 5 points
    analogWrite(dimLed, fadeValue); //sets the value (range from 0 to 255):
    delay(20); //wait for 30 milliseconds to see the dimming effect

    }
    //check if the pushbutton is pressed. If not, buttonValue is low and turn off the dim LED.
  } else {
    digitalWrite(dimLed, LOW);
  }
}
